function upsert(ip, data) {
  const key = data.hostname || 'unknown';
  const cleanIp = (ip||'').replace(/^::ffff:/,'');
  _employeeIPs.add(cleanIp);

  const existing = reg.get(key) || {};

  if (data.type==='backfill' && data.windows) {
    const existingTs=new Set((existing.timeline||[]).map(w=>w.ts));
    const newWins=data.windows.filter(w=>!existingTs.has(w.ts));
    const timeline=[...(existing.timeline||[]),...newWins].sort((a,b)=>new Date(a.ts)-new Date(b.ts));
    log('BACKFILL',`Merged ${newWins.length} windows for ${key}`);
    reg.set(key,{...existing,timeline,lastSeen:Date.now()});
    return;
  }

  // Merge EOD timeline with any existing windows (never wipe accumulated data)
  let mergedTimeline = existing.timeline || [];
  if (data.type === 'eod' && data.timeline && data.timeline.length) {
    const existingTs = new Set(mergedTimeline.map(w => w.ts));
    const newWins = data.timeline.filter(w => !existingTs.has(w.ts));
    mergedTimeline = [...mergedTimeline, ...newWins].sort((a, b) => new Date(a.ts) - new Date(b.ts));
    log('EOD', `Merged ${newWins.length} EOD windows for ${key}, total=${mergedTimeline.length}`);
  }

  reg.set(key, {
    hostname:key, ip:cleanIp, platform:data.platform||'?',
    lastSeen:Date.now(), lastType:data.type, state:data.state,
    signals:{
      state:data.state, idle_s:data.idle_s, active_win:data.active_win,
      screen_locked:data.screen_locked, processes:data.processes,
      login_time:data.login_time, spoof_score:data.spoof_score,
      spoof_reasons:data.spoof_reasons, prod_score:data.prod_score,
    },
    summary:data.summary||existing.summary||null,
    timeline:mergedTimeline,
    alerts:data.alerts||[],
  });

  (data.alerts||[]).forEach(a=>store.addAlert({...a,hostname:key}));

  if (data.type==='lifecycle'&&data.event) {
    store.addAlert({
      id:Date.now().toString(36),hostname:key,type:data.event,
      message:`${key} ${data.event} at ${new Date(data.ts).toLocaleTimeString()}`,
      severity:'info',ts:data.ts||new Date().toISOString(),dismissed:false,
    });
    log('LIFECYCLE',`${key}: ${data.event}`);
  }

  if (data.type==='site_alert') {
    (data.alerts||[]).forEach(a=>{
      if(a.type==='blocked_site'){
        store.addAlert({...a,id:Date.now().toString(36),hostname:key,dismissed:false});
        log('SITE',`${key}: ${a.category} â€” ${a.site}`);
      }
    });
  }
}

function _updateSysCache() {
  try {
    // CPU/RAM from Node os module — instant, no blocking
    const cpus1 = _cpus0;
    const cpus2 = os.cpus();
    _cpus0 = cpus2;
    let idle=0, total=0;
    cpus2.forEach((c,i)=>{
      const d1=cpus1[i]?cpus1[i].times:c.times, d2=c.times;
      idle  += d2.idle - d1.idle;
      total += (d2.user+d2.nice+d2.sys+d2.idle+d2.irq)-(d1.user+d1.nice+d1.sys+d1.idle+d1.irq);
    });
    const cpuPct = total>0 ? Math.max(0,Math.min(100,Math.round(100*(1-idle/total)))) : 0;
    const totalMem=os.totalmem(), freeMem=os.freemem();
    const ramPct=Math.round(((totalMem-freeMem)/totalMem)*100);
    const uptime=os.uptime();
    _sysCache = {
      cpu:cpuPct, ram:ramPct,
      ramUsedGB:parseFloat(((totalMem-freeMem)/1024/1024/1024).toFixed(1)),
      ramTotalGB:parseFloat((totalMem/1024/1024/1024).toFixed(1)),
      cores:cpus2.length, model:(cpus2[0]&&cpus2[0].model)||'',
      uptimeH:Math.floor(uptime/3600), uptimeM:Math.floor((uptime%3600)/60),
      ts:Date.now(), procs:_sysCache.procs||[]
    };
    if(cpuPct>85) store.addAlert({id:'cpu-'+Date.now().toString(36),hostname:os.hostname(),
      type:'high_cpu',message:`High CPU: ${cpuPct}% on boss PC`,
      severity:'danger',ts:new Date().toISOString(),dismissed:false});
    if(ramPct>90) store.addAlert({id:'ram-'+Date.now().toString(36),hostname:os.hostname(),
      type:'high_ram',message:`High RAM: ${ramPct}% on boss PC`,
      severity:'danger',ts:new Date().toISOString(),dismissed:false});
  } catch(_) {}
  // Fetch process list asynchronously — non-blocking
  _updateProcsAsync();
}

function _updateProcsAsync() {
  try {
    const {spawn} = require('child_process');
    const ps = spawn('powershell', [
      '-NoProfile','-NonInteractive','-Command',
      "$p=Get-Process|Where-Object{$_.CPU -gt 0}|Sort-Object CPU -Desc|Select-Object -First 15 Name,Id,@{N='C';E={[math]::Round($_.CPU/((Get-Date)-$_.StartTime).TotalSeconds,1)}},@{N='M';E={[math]::Round($_.WorkingSet64/1MB,1)}};$p|ForEach-Object{Write-Output ('PROC:'+$_.Name+'|'+$_.Id+'|'+$_.C+'|'+$_.M)}"
    ], {timeout:8000, stdio:['ignore','pipe','ignore']});
    let out='';
    ps.stdout.on('data', d => { out += d.toString(); });
    ps.on('close', () => {
      try {
        const procs = out.split('\n').map(l=>l.trim()).filter(l=>l.startsWith('PROC:')).map(l=>{
          const p=l.slice(5).split('|');
          return {name:p[0],pid:parseInt(p[1])||0,cpu:parseFloat(p[2])||0,mem:parseFloat(p[3])||0};
        });
        _sysCache = {..._sysCache, procs};
      } catch(_) {}
    });
    ps.on('error', () => {});
  } catch(_) {}
}

// â”€â”€ Summary cache â€” avoids recalculating daySummary on every API request â”€â”€â”€â”€â”€â”€
let _sumCache = null;
let _sumCacheTs = 0;
const SUM_CACHE_TTL = 30000; // recalculate at most every 30s

function getSelfSum() {
  const now = Date.now();
  if (_selfSum && _sumCache && (now - _sumCacheTs) < SUM_CACHE_TTL) return _sumCache;
  _sumCache = _selfSum;
  _sumCacheTs = now;
  return _sumCache;
}

function startSampling() {
  act.init();
  function tick() {
    _selfSnap = act.snapshot();
    const s = new Date(); s.setHours(0,0,0,0);
    const e = new Date(); e.setHours(23,59,59,999);
    let sum = act.daySummary(s.toISOString(), e.toISOString());
    // Fallback: timezone offset handling â€” try last 24h window
    if (!sum) {
      const s2 = new Date(Date.now() - 24*3600*1000);
      sum = act.daySummary(s2.toISOString(), e.toISOString());
    }
    _selfSum = sum;
    _sumCache = sum;       // update cache on each tick
    _sumCacheTs = Date.now();
  }
  tick(); setInterval(tick, C.SAMPLE_MS);
  // Keep boss entry in reg for timeline/shift endpoints
  setInterval(function() {
    const key = os.hostname();
    const existing = reg.get(key) || {};
    // Only keep today's timeline windows in reg â€” reduces memory per request
    const cutoff = new Date(); cutoff.setHours(0,0,0,0);
    const todayTl = act.getTimeline().filter(w => new Date(w.ts) >= cutoff);
    reg.set(key, {...existing, hostname:key, timeline:todayTl, lastSeen:Date.now()});
  }, C.SAMPLE_MS);
}

function startDiscovery() {
  const sock=dgram.createSocket({type:'udp4',reuseAddr:true});
  sock.on('error',e=>log('UDP',e.message));
  sock.on('message',(msg,r)=>{
    const msgStr = msg.toString().trim();
    if(msgStr===C.DISCOVERY_MSG){
      sock.send(Buffer.from(C.DISCOVERY_ACK),r.port,r.address);
      log('DISC','Discovery from '+r.address+' - ACK sent');
    } else {
      // Also accept JSON heartbeat/signal packets from child agents
      try {
        const data = JSON.parse(msgStr);
        if(data.hostname){
          upsert(r.address, data);
          log('UDP','Signal from '+r.address+' ('+data.hostname+') state='+data.state);
        }
      } catch(_) {}
    }
  });
  sock.bind(C.DISCOVERY_PORT,()=>log('UDP',':'+C.DISCOVERY_PORT));
}

    if (req.method==='POST'&&url==='/api/kill-top-proc'){
      let b='';req.on('data',c=>b+=c);req.on('end',()=>{
        try{
          const d=JSON.parse(b||'{}');

          // Find the employee in registry and get their sysmetrics
          // Then send a kill command via /api/kill-process on their local server

          const ps=spawn('powershell',['-NoProfile','-NonInteractive','-Command',
            "Get-Process|Sort-Object CPU -Desc|Select-Object -First 1 Name,Id|ForEach-Object{Write-Output ($_.Name+','+$_.Id)}"
          ],{stdio:['ignore','pipe','ignore'],windowsHide:true,timeout:5000});

          ps.stdout.on('data',chunk=>{out+=chunk.toString();});
          ps.on('close',()=>{
            try{
              const parts=out.trim().split(',');

              if(!procName) return json(res,500,{ok:false,error:'No process found'});

          _bossReplies[hostname].push({

    }

    if (req.method==='GET'&&url==='/api/boss-replies'){

        // Past midnight â€” overnight shift started yesterday
        baseDate = new Date(now); baseDate.setDate(baseDate.getDate()-1);
      } else if (!isOvernight) {
        // Day shift: if we are currently in an overnight shift (localHour >= nightStart or localHour < nightEnd),
        // show the most recent completed day shift (yesterday)
        const nightShift = shifts.find(s => s.end <= s.start);
        const inNight = nightShift && (localHour >= nightShift.start || localHour < nightShift.end);
        if (inNight) {
          // Currently in night shift â€” show yesterday's day shift data
          baseDate = new Date(now); baseDate.setDate(baseDate.getDate()-1);
        } else {
          baseDate = new Date(now);
        }
      } else {
        baseDate = new Date(now);
      }
      // Build ss/se as local-time Date objects
      // Build shift window in local wall-clock time.
      // new Date(y,m,d,h) uses the process local timezone â€” on Windows this
      // is the system timezone (CDT in Milwaukee), so it matches getHours().
      // We convert to UTC-equivalent by using Date.UTC and subtracting tzOffset.
      const tzOff = now.getTimezoneOffset(); // minutes behind UTC (e.g. 300 for CDT)
      function localHMS(base, h, m, s) {
        // Create a date that represents base-date at h:m:s in LOCAL time
        // by constructing in UTC then shifting by tzOffset
        const d = new Date(Date.UTC(base.getFullYear(), base.getMonth(), base.getDate(), h, m||0, s||0));
        d.setMinutes(d.getMinutes() + tzOff);
        return d;
      }

        se = localHMS(baseDate, shift.end);

        se = localHMS(nextDay, shift.end);

      const filtered=tl.filter(w=>{const t=new Date(w.ts);return t>=ss&&t<se;}); log("SHIFT","ss="+ss.toISOString()+" se="+se.toISOString()+" tl="+tl.length+" filtered="+filtered.length);

        start:ss.toISOString(),end:se.toISOString(),
        windows:total,
        is_yesterday:baseDate.toDateString()!==now.toDateString(),
        active_minutes:Math.round(active*10/60),
        idle_minutes:Math.round(idle*10/60),
        away_minutes:Math.round(away*10/60),
        locked_minutes:Math.round(locked*10/60),
        net_active_minutes:Math.round((active*10/60)),
        shift_total_minutes:(shift.end>shift.start?(shift.end-shift.start):((24-shift.start)+shift.end))*60,
        productive_pct:prod,
      });
    }
    if (url==='/api/timeline') {
      let tl;
      if (qs.host==='__boss__') {

  });

  });
  srv.on('error',e=>log('PARENT','Error: '+e.message));
}
